  <script>
    function onSearch(e){
      e.preventDefault();
      const q = (e.target.q && e.target.q.value) || document.querySelector('.search-input').value;
      if(!q.trim()) return;
      // En este mockup simulamos la búsqueda redirigiendo a Google real.
      window.location.href = 'https://www.google.com/search?q=' + encodeURIComponent(q);
    }
